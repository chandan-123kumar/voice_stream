# Python dependencies for the Qwen3-TTS megakernel project.
# Versions pinned to what was tested on the RTX 5090 box (2026-06).
#
# NOT listed here on purpose:
#   - torchaudio: must stay as the stub — stock wheels are ABI-incompatible
#     with this torch build. If any install below pulls torchaudio in,
#     run `bash scripts/fix_torchaudio_stub.sh` afterwards (see SETUP.md).
#   - torchvision: do NOT install. Nothing here uses it, but if present and
#     built against a different CUDA than torch, transformers auto-imports it
#     and `AutoProcessor` breaks. The fix script removes a mismatched one.

# Stock PyPI torch works on the RTX 5090 (the cu128 wheel ships sm_120).
# Pinned because the container's prerelease build (2.10.0a0+nv26.1) does NOT
# satisfy pip's resolver — on 2026-06-10 `pip install -r requirements.txt`
# silently replaced it with this stock wheel, which was then verified to run
# the megakernel at full speed. Pinning makes that explicit and repeatable.
torch==2.10.0

qwen-tts==0.1.1          # official Qwen3-TTS package (model, codec, prompts)
soundfile==0.13.1        # wav read/write in tests and CLI
numpy==2.1.0             # already in the container; pinned for reproducibility

# Web app (app.py) — these usually arrive as qwen-tts transitive deps,
# listed explicitly since app.py imports them directly:
fastapi==0.136.3
uvicorn==0.49.0
pydantic==2.12.5

# Voice agent demo (voice_agent.py) + in-process Pipecat TTS service:
pipecat-ai==1.3.0        # pipeline framework (transport, VAD, turn-taking)
websockets==16.0         # pipecat websocket transport dependency
python-dotenv==1.2.2     # loads OPENAI_API_KEY from .env
faster-whisper~=1.2.1    # local GPU STT (STT_BACKEND=whisper, the default)

# Build-time:
ninja==1.13.0            # used by torch cpp_extension to JIT-compile the kernel
